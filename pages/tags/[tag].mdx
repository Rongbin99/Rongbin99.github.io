import { useRouter } from "next/router";
import Lost from '../../components/lost.js';

export const TagName = () => {
  const { tag } = useRouter().query;

  if (tag == "side_project") {
    return "Side Projects";
  } else if (tag || null) {
    return `Posts Tagged with "${tag}"`;
  } else {
    <Lost />
    return null;
  }
};

# <TagName />
